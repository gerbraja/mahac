```python
        if (member.registered_at - upline_registered_at).days <= 367:
            # Verificar que est√© activo o no haya expirado
            if member.is_active:
                active_members.append(member)
            elif (today - member.registered_at).days < 120:
                # Pre-afiliado a√∫n v√°lido (no han pasado 120 d√≠as)
                active_members.append(member)
    
    return len(active_members)

def clean_expired_members():
    """
    Tarea CRON que se ejecuta diariamente.
    Elimina miembros pre-afiliados que no se activaron en 120 d√≠as.
    Comprime el √°rbol autom√°ticamente.
    """
    expired_members = get_members_where(
        is_active=False,
        days_since_registration__gt=120
    )
    
    for member in expired_members:
        # Obtener hijos del miembro expirado
        children = get_downlines(member.id)
        
        # Reasignar hijos al upline del miembro expirado
        for child in children:
            child.upline_id = member.upline_id
            save(child)
        
        # Eliminar miembro expirado
        delete(member)
        
        log(f"Eliminado miembro {member.user_id} por expiraci√≥n")
```
