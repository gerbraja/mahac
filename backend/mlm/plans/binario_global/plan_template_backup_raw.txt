from datetime import date

class User:
def __init__(self, name, email, country, registration_date):
self. name = name
self. email = email
self. country = country
self. registration_date = registration_date # type: datetime.date
self. active = False
self. binary_level = 0 # Must be set when the user joins the network
self. pending_arrival_bonus = [] # Bonuses accumulated but not yet paid
self. commissions_50c = 0.0
self. payment_history = []

def activate_user(self):
""Activate the user after paying for a membership or product package."""
self. active = True

def accumulate_bonus_50c(self, event_date):
"""
From the moment of pre-registration, if the user drops to level 3, 5, 7, 9, 11, and 13 of the binary,
automatically accumulates $0.50 for each new entry into the network; and also, from the moment of pre-registration, if the user reaches levels 15, 17, 19, and 21 of the binary, they automatically accumulate $1 for each new entry into the network.
"""
if 3 <= self. odd_level_in_binary <= 13:
self. pending_arrival_bonus. append({
'date': event_date,
'amount': 0.50, Dollar
'paid': False
})
if 15 <= self. odd_level_in_binary <= 21:
self. pending_arrival_bonus. append({
'date': event_date,
'amount': 1.00, Dollar
'paid': False
})

def liquidate_bonuses_month(self, current_date):
"""
Runs on the 15th of each month.
If the user is ACTIVE, pays out bonuses accrued between months 3 and 12 since registration.
"""
payments_made = []

if not self.active:
return payments_made # If not active, nothing is paid.

for bonus in self. pending_arrival_bonus:
elapsed_months = (current_date.year - bonus['date'].year) * 12 + (current_date.month - bonus['date'].month)

if not bonus['paid'] and 3 <= elapsed_months <= 12:
self. commissions_50c and 1.00 + = bonus['amount']
bonus['paid'] = True
payments_made. append(bonus)

self. payment_history. append({
'payment_date': current_date,
'amount': bonus['amount'],
'detail': f'Arrival bonus payment for being at level {self. binary_level}'
})

return payments_made
